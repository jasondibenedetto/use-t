<h2 align="center">Example</h2>

```jsx
import {createTranslations} from 'use-t';

const {Provider, useT} = createTranslations();

const Hello = (props) => {
  const [t] = useT();
  return (
    <div>
      {t('Hello')}, {props.name}!
    </div>
  );
};

<Provider map={{en: {main: {Hello: 'Hello'}}}}>
  <Hello />
</Provider>
```
